# unit_converter.py # A simple Python script for common engineering unit conversions. # This can be expanded with more conversions as needed. # Usage: python unit_converter.py --type --value # Example: python unit_converter.py --type mm_to_inches --value 25.4 import argparse def cmh_to_cfm(cmh): """Convert cubic meters per hour (CMH) to cubic feet per minute (CFM).""" return cmh * 0.58858 def mm_to_inches(mm): """Convert millimeters (mm) to inches.""" return mm / 25.4 def kw_to_btuh(kw): """Convert kilowatts (kW) to BTU per hour (BTUH).""" return kw * 3412.142 def celsius_to_fahrenheit(c): """Convert Celsius to Fahrenheit (optional additional conversion).""" return (c * 9/5) + 32 def meters_to_feet(m): """Convert meters to feet (optional additional conversion).""" return m * 3.28084 def pa_to_psi(pa): """Convert Pascals (Pa) to pounds per square inch (PSI) (optional).""" return pa * 0.000145038 # Dictionary of available conversions conversions = { 'cmh_to_cfm': cmh_to_cfm, 'mm_to_inches': mm_to_inches, 'kw_to_btuh': kw_to_btuh, 'celsius_to_fahrenheit': celsius_to_fahrenheit, 'meters_to_feet': meters_to_feet, 'pa_to_psi': pa_to_psi, } def main(): parser = argparse.ArgumentParser(description="Engineering Unit Converter") parser.add_argument('--type', required=True, choices=conversions.keys(), help="Type of conversion") parser.add_argument('--value', required=True, type=float, help="Value to convert") args = parser.parse_args() converter = conversions[args.type] result = converter(args.value) print(f"{args.value} {args.type.split('_to_')[0]} = {result:.4f} {args.type.split('_to_')[1]}") if __name__ == "__main__": main()
