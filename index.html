/* ============================================================ GLOBAL HVAC SOLUTIONS — COMPLETE ENGINE FILE Airflow • Refrigerant PT • Psychrometrics (with Dew Point) ============================================================ */ /* ------------------------------------------------------------ 1. AIRFLOW — CFM / VELOCITY / AREA / DIAMETER ------------------------------------------------------------ */ // CFM = Area × Velocity function cfmFromAreaVelocity(areaSqFt, velocityFpm) { return areaSqFt * velocityFpm; } function velocityFromCfmArea(cfm, areaSqFt) { return cfm / areaSqFt; } function areaFromCfmVelocity(cfm, velocityFpm) { return cfm / velocityFpm; } // Convert area (sq ft) → equivalent round diameter (inches) function diameterInchesFromAreaSqFt(areaSqFt) { const areaSqIn = areaSqFt * 144; return Math.sqrt((4 * areaSqIn) / Math.PI); // Correct formula } // Convert diameter (inches) → area (sq ft) function areaSqFtFromDiameterInches(dIn) { const areaSqIn = (Math.PI * dIn * dIn) / 4; return areaSqIn / 144; } // Standard round duct sizes const STANDARD_ROUND_SIZES_IN = [4,5,6,7,8,9,10,11,12,14,16,18,20]; function nearestStandardRoundDiameter(dIn) { let best = STANDARD_ROUND_SIZES_IN[0]; let bestDiff = Math.abs(dIn - best); for (const size of STANDARD_ROUND_SIZES_IN) { const diff = Math.abs(dIn - size); if (diff < bestDiff) { best = size; bestDiff = diff; } } return best; } // Ductulator-style sizing: CFM + target velocity function sizeDuctFromCfmAndVelocity(cfm, targetVelocityFpm, snap = true) { const areaSqFt = areaFromCfmVelocity(cfm, targetVelocityFpm); const rawD = diameterInchesFromAreaSqFt(areaSqFt); const dIn = snap ? nearestStandardRoundDiameter(rawD) : rawD; const actualArea = areaSqFtFromDiameterInches(dIn); const actualVelocity = velocityFromCfmArea(cfm, actualArea); return { rawDiameterIn: rawD, diameterIn: dIn, areaSqFt: actualArea, velocityFpm: actualVelocity }; } /* ------------------------------------------------------------ 2. REFRIGERANT PRESSURE ↔ TEMPERATURE (TP CHART) ------------------------------------------------------------ */ /* IMPORTANT: Replace the sample values below with YOUR actual TP chart values so the app matches your field charts exactly. */ const REFRIG_PT_TABLES = { R410A: [ { p: 50, t: 16 }, { p: 100, t: 31 }, { p: 150, t: 43 }, { p: 200, t: 54 }, { p: 250, t: 63 }, { p: 300, t: 72 } ], R22: [ { p: 50, t: 26 }, { p: 75, t: 35 }, { p: 100, t: 41 }, { p: 125, t: 47 }, { p: 150, t: 52 }, { p: 175, t: 57 } ] }; // Linear interpolation function interpolatePT(table, psig) { if (!table || table.length === 0) return null; if (psig <= table[0].p) return table[0].t; if (psig >= table[table.length - 1].p) return table[table.length - 1].t; for (let i = 0; i < table.length - 1; i++) { const p1 = table[i].p; const t1 = table[i].t; const p2 = table[i + 1].p; const t2 = table[i + 1].t; if (psig >= p1 && psig <= p2) { const ratio = (psig - p1) / (p2 - p1); return t1 + ratio * (t2 - t1); } } return null; } function satTempFromPressure(refrigerantKey, psig) { const table = REFRIG_PT_TABLES[refrigerantKey]; return interpolatePT(table, psig); } /* ------------------------------------------------------------ 3. PSYCHROMETRICS — DEW POINT (CORRECTED + DISPLAY) ------------------------------------------------------------ */ // Dew point from dry-bulb (°F) and RH (%) function dewPointF(dbF, rhPercent) { const dbC = (dbF - 32) * 5/9; const rh = rhPercent / 100; const a = 17.27; const b = 237.7; const gamma = Math.log(rh) + (a * dbC) / (b + dbC); const dewC = (b * gamma) / (a - gamma); return dewC * 9/5 + 32; } // UI output block for psychrometrics function showPsychrometricsResults(dbF, rhPercent) { const dewF = dewPointF(dbF, rhPercent); const resultHTML = `

Dry‑bulb: ${dbF.toFixed(1)} °F

Relative Humidity: ${rhPercent.toFixed(1)} %

Dew Point: ${dewF.toFixed(1)} °F

`; document.getElementById("psychResults").innerHTML = resultHTML; }
