def cfm_to_m3h(val):    return val * 1.69901
def m3h_to_cfm(val):    return val / 1.69901
def mm_to_inches(val):  return val / 25.4
def inches_to_mm(val):  return val * 25.4
def kw_to_btuh(val):    return val * 3412.142
def btuh_to_kw(val):    return val / 3412.142
def btuh_to_tons(val):  return val / 12000
def tons_to_btuh(val):  return val * 12000
def c_to_f(val):        return val * 9/5 + 32
def f_to_c(val):        return (val - 32) * 5/9
def m_to_ft(val):       return val * 3.28084
def pa_to_inwg(val):    return val * 0.00401463
