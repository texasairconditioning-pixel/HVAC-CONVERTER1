# unit_converter.py (improved version) import argparse # Conversion functions def cmh_to_cfm(val): return val * 0.588578 def cfm_to_cmh(val): return val / 0.588578 def mm_to_inches(val): return val / 25.4 def inches_to_mm(val): return val * 25.4 def kw_to_btuh(val): return val * 3412.142 def btuh_to_kw(val): return val / 3412.142 def c_to_f(val): return val * 9/5 + 32 def f_to_c(val): return (val - 32) * 5/9 def m_to_ft(val): return val * 3.28084 def pa_to_inwg(val): return val * 0.00401463 # Pa → inches water gauge CONVERSIONS = { 'cmh_to_cfm': (cmh_to_cfm, 'CMH', 'CFM'), 'cfm_to_cmh': (cfm_to_cmh, 'CFM', 'CMH'), 'mm_to_inches': (mm_to_inches, 'mm', 'inches'), 'inches_to_mm': (inches_to_mm, 'inches', 'mm'), 'kw_to_btuh': (kw_to_btuh, 'kW', 'BTU/h'), 'btuh_to_kw': (btuh_to_kw, 'BTU/h', 'kW'), 'c_to_f': (c_to_f, '°C', '°F'), 'f_to_c': (f_to_c, '°F', '°C'), 'm_to_ft': (m_to_ft, 'm', 'ft'), 'pa_to_inwg': (pa_to_inwg, 'Pa', 'in. wg'), } def list_conversions(): print("Available conversions:") for key, (_, from_unit, to_unit) in sorted(CONVERSIONS.items()): print(f" {key:16} {from_unit:>6} → {to_unit}") def main(): parser = argparse.ArgumentParser( description="Engineering unit converter (HVAC, mechanical, etc.)", epilog="Example: python unit_converter.py --type kw_to_btuh --value 5.5" ) parser.add_argument('-t', '--type', help="Conversion type (use --list to see all)") parser.add_argument('-v', '--value', type=float, help="Numeric value to convert") parser.add_argument('--list', action='store_true', help="Show all available conversions and exit") args = parser.parse_args() if args.list: list_conversions() return if not args.type or args.value is None: parser.print_help() return conv_key = args.type.lower() if conv_key not in CONVERSIONS: print(f"Error: Unknown conversion '{args.type}'") print("Run with --list to see available options.") return func, from_u, to_u = CONVERSIONS[conv_key] result = func(args.value) print(f"{args.value:g} {from_u} = {result:,.2f} {to_u}") print(f" (exact: {result:.6f} {to_u})") if __name__ == "__main__": main()
