/* ============================================================
   INTERPOLATION — TEMP → PRESSURE
   ============================================================ */
function interpolateTempToPressure(ref, tempF) {
    const data = PT_DATA[ref];
    const T = data.temps;
    const P = data.press;

    if (tempF <= T[0]) return P[0];
    if (tempF >= T[T.length - 1]) return P[P.length - 1];

    for (let i = 0; i < T.length - 1; i++) {
        if (tempF >= T[i] && tempF <= T[i + 1]) {
            const ratio = (tempF - T[i]) / (T[i + 1] - T[i]);
            return P[i] + ratio * (P[i + 1] - P[i]);
        }
    }
}

/* ============================================================
   REVERSE INTERPOLATION — PRESSURE → TEMP
   ============================================================ */
function interpolatePressureToTemp(ref, psi) {
    const data = PT_DATA[ref];
    const T = data.temps;
    const P = data.press;

    if (psi <= P[0]) return T[0];
    if (psi >= P[P.length - 1]) return T[T.length - 1];

    for (let i = 0; i < P.length - 1; i++) {
        if (psi >= P[i] && psi <= P[i + 1]) {
            const ratio = (psi - P[i]) / (P[i + 1] - P[i]);
            return T[i] + ratio * (T[i + 1] - T[i]);
        }
    }
}

/* ============================================================
   R3 — FULL REFRIGERATION LOGIC + UI INTEGRATION
   ============================================================ */

/* ------------------------------------------------------------
   1. TEMP → PRESSURE
   ------------------------------------------------------------ */
function calcTempToPressure() {
    const ref = document.getElementById("refType").value;
    const evapT = parseFloat(document.getElementById("evapTemp").value);
    const condT = parseFloat(document.getElementById("condTemp").value);

    let out = "";

    if (!isNaN(evapT)) {
        const evapP = interpolateTempToPressure(ref, evapT);
        out += `Evaporator Pressure: ${evapP.toFixed(1)} PSIG\n`;
        last_evapP = evapP;
    }

    if (!isNaN(condT)) {
        const condP = interpolateTempToPressure(ref, condT);
        out += `Condenser Pressure: ${condP.toFixed(1)} PSIG\n`;
        last_condP = condP;
    }

    if (out === "") out = "Enter at least one temperature.";

    document.getElementById("tempToPressResult").innerText = out;
    updateSummary();
}

/* ------------------------------------------------------------
   2. PRESSURE → TEMP
   ------------------------------------------------------------ */
function calcPressureToTemp() {
    const ref = document.getElementById("refType").value;
    const sucP = parseFloat(document.getElementById("suctionPSI").value);
    const liqP = parseFloat(document.getElementById("liquidPSI").value);

    let out = "";

    if (!isNaN(sucP)) {
        const evapT = interpolatePressureToTemp(ref, sucP);
        out += `Evaporator Saturation Temp: ${evapT.toFixed(1)} °F\n`;
        last_evapT = evapT;
    }

    if (!isNaN(liqP)) {
        const condT = interpolatePressureToTemp(ref, liqP);
        out += `Condenser Saturation Temp: ${condT.toFixed(1)} °F\n`;
        last_condT = condT;
    }

    if (out === "") out = "Enter at least one pressure.";

    document.getElementById("pressToTempResult").innerText = out;
    updateSummary();
}

/* ------------------------------------------------------------
   3. SUPERHEAT
   ------------------------------------------------------------ */
function calcSuperheat() {
    const ref = document.getElementById("refType").value;
    const sucP = parseFloat(document.getElementById("shSuction").value);
    const lineT = parseFloat(document.getElementById("shTemp").value);

    if (isNaN(sucP) || isNaN(lineT)) {
        document.getElementById("superheatResult").innerText =
            "Enter suction pressure and suction line temperature.";
        return;
    }

    const satT = interpolatePressureToTemp(ref, sucP);
    const SH = lineT - satT;

    last_evapT = satT;
    last_SH = SH;

    document.getElementById("superheatResult").innerText =
        `Saturation Temp: ${satT.toFixed(1)} °F\n` +
        `Superheat: ${SH.toFixed(1)} °F`;

    updateSummary();
}

/* ------------------------------------------------------------
   4. SUBCOOLING
   ------------------------------------------------------------ */
function calcSubcooling() {
    const ref = document.getElementById("refType").value;
    const liqP = parseFloat(document.getElementById("scLiquid").value);
    const lineT = parseFloat(document.getElementById("scTemp").value);

    if (isNaN(liqP) || isNaN(lineT)) {
        document.getElementById("subcoolResult").innerText =
            "Enter liquid pressure and liquid line temperature.";
        return;
    }

    const satT = interpolatePressureToTemp(ref, liqP);
    const SC = satT - lineT;

    last_condT = satT;
    last_SC = SC;

    document.getElementById("subcoolResult").innerText =
        `Saturation Temp: ${satT.toFixed(1)} °F\n` +
        `Subcooling: ${SC.toFixed(1)} °F`;

    updateSummary();
}

/* ------------------------------------------------------------
   5. SUMMARY BUILDER
   ------------------------------------------------------------ */
let last_evapT = null;
let last_condT = null;
let last_evapP = null;
let last_condP = null;
let last_SH = null;
let last_SC = null;

function updateSummary() {
    const ref = document.getElementById("refType").value;

    let out = `Refrigerant: ${ref}\n\n`;

    if (last_evapT !== null) out += `Evap Sat Temp: ${last_evapT.toFixed(1)} °F\n`;
    if (last_evapP !== null) out += `Evap Pressure: ${last_evapP.toFixed(1)} PSIG\n`;

    if (last_condT !== null) out += `Cond Sat Temp: ${last_condT.toFixed(1)} °F\n`;
    if (last_condP !== null) out += `Cond Pressure: ${last_condP.toFixed(1)} PSIG\n`;

    if (last_SH !== null) out += `Superheat: ${last_SH.toFixed(1)} °F\n`;
    if (last_SC !== null) out += `Subcooling: ${last_SC.toFixed(1)} °F\n`;

    document.getElementById("summaryOutput").innerText = out;
}
